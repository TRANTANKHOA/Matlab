1%   ADAPTIVE METROPOLIS: Generate random samples from a target density phi
  2%   with a Gaussian proposal. Function syntax is:
  3%
  4%   G = AM(X,V,M,OPT)
  5%  
  6%   in which:
  7%
  8%   X:           Initial sample
  9%   V:           Initial covariance matrix to approximate the geometry of 
 10%                the target distribution
 11%   M:           Number of samples to be generated
 12%   OPT:
 13%   OPT.data     Necessary data to compute the density function  
 14%   OPT.dsp      Display completion status
 15%   OPT.hgrd     Number of grid points in histograms, must be multiple of 10
 16%   OPT.filename Naming the file to store MCMC samples
 17%
 18%
 19%
 20%   Written by Khoa T. Tran, School of EE & CS
 21%                            University of Newcastle
 22%                            Australia.
 23%
 24%   Copyright (C) Khoa T. Tran
 25%
 26%   Definitions of additional terms in the codes
 27%   ---------------------------------
 28%   logpdf    % Compute the log of the density function. To facilitate
 29%               Matlab coder translation, this function is to be hardcoded
 30%               by the user in a separate logpdf.m file.
 31%   burnin    % Burn-in time
 32%   best_X    % Sample with highest density
 33%   best_P    % Highest density
 34%   X_bar     % Sample mean
 35%   V         % Empirical Covariance Matrix
 36%   delta     % Historical adaptation steps
 37%   s         % Metropolis scaling factor
 38%   acc_rate  % Metropolis acceptance rate
 39
 40function G = am(X,V,M,OPT)
 41%% Initiation
 42coder.extrinsic('warning','cputime','num2str','save');
 43Dims        = length(X);            % Number of dimensions
 44best_X      = X;                    % Record highest density state
 45P0          = logpdf(X,OPT.data);   % Last density
 46best_P      = P0;                   % Best seen density
 47eps         = -0.8;                 % Could be -0.5 to -1
 48pcom        = 0;                    % Percentage complete count initialised to zero;
 49chk_stop    = 2000;                 % Periodically check to stop burn-in
 50burnin      = M;                    % Initial burnin set to 100% sample
 51delta       = zeros(burnin,2);      % Step change in adaptation
 52% Take the Cholesky factor of V
 53[V2,errV]=chol(V);
 54if errV
 55    warning('Bad covariance matrix at start, perform reconditioning..')
 56    [V3,D] = eig(V);d=diag(D);d(d<=0)=10*realmin;
 57    V= V3*diag(d)*V3';
 58    L = tril(V,-1);V=diag(diag(V))+L+L';
 59    V2=chol(V);
 60end
 61% Define optimal acceptance rate
 62switch Dims 
 63    case 1
 64        optimal=0.441;
 65    case 2
 66        optimal=0.352;
 67    case 3
 68        optimal=0.316;
 69    case 4
 70        optimal=0.285;
 71    case 5
 72        optimal=0.275;
 73    case 6
 74        optimal=0.273;
 75    case 7
 76        optimal=0.270;
 77    case 8
 78        optimal=0.268;
 79    case 9
 80        optimal=0.267;
 81    case 10
 82        optimal=0.266;
 83    case 11
 84        optimal=0.265;
 85    case 12
 86        optimal=0.264;
 87    otherwise
 88        optimal=0.255;
 89end
 90% Set initial scaling factor
 91s = 2.38^2/Dims;
 92%% Adaptive Metropolis Algorithm.
 93U = log(rand(burnin,1));        % Random acceptance numbers 
 94RD = randn(burnin,Dims);        % Random distubance proposal
 95prerun = 100*Dims;              % Generate 'prerun' samples before burnin
 96X_2 = zeros(prerun,Dims);       % Pre-burnin samples
 97avg_acc = 0;                    % Average acceptance rate
 98%% Pre-burnin run - generates 'prerun' samples with the initial covariance matrix
 99mark = cputime;mark_0 = cputime;% Used to keep track of elapsed time
100for i = 1:prerun
101    if OPT.dsp % Display completion status
102        if mod(i,floor(burnin/20))==0
103            pcom=pcom+5;
104            fprintf('Complete = %d%% burn-in, Time since last update = %f s\n',pcom,cputime-mark); 
105            if i>2
106            remaining = (20 - i/floor(burnin/20))*(cputime-mark);
107            hrs = floor(remaining/3600); remaining = rem(remaining,3600);
108            mins = floor(remaining/60); 
109            secs = floor(rem(remaining,60));
110            fprintf('Predicted burn-in completion in %d hrs:%d mins:%d secs\n',hrs,mins,secs)
111            fprintf('Acceptance rate %f %%\n',100*avg_acc);
112            end;
113        mark=cputime;
114        end;
115    end; 
116    X_new = X + sqrt(s)*RD(i,:)*V2; % propose a new sample
117    P_new = logpdf(X_new,OPT.data);          % new density
118    if P_new>best_P                 % new density is highest 
119        best_X=X_new;               % record best seen parameters
120        best_P = P_new;
121        X = X_new;                  % accept X_new 
122        P0 = P_new;
123        acc_rate=1;                 % record acceptance rate
124    else 
125        rho = P_new - P0;
126        acc_rate=exp(min(rho,0));   % record acceptance rate
127        if U(i)<= rho
128        X = X_new;                  % accept X_new 
129        P0 = P_new;
130        end
131    end
132    X_2(i,:) = X;                   % Store prerun samples
133    %Update scale factor by Robbins-Monro approximation and record everage acceptance rate
134    delta(i,1) = (i^eps)*(acc_rate-optimal);
135    s = exp(log(s)+delta(i,1));
136    avg_acc = avg_acc*(i-1)/i + acc_rate/i;
137end
138X_bar=mean(X_2);          % Sample mean
139if OPT.cov == 1
140    V=cov(X_2);                % Sample covariance matrix
141    L = tril(V,-1);V=diag(diag(V))+L+L'; % Check for symmetry
142    [V2,errV]=chol(V);
143    if errV
144        warning('Bad covariance matrix at prerun, perform reconditioning..')
145        [V3,D] = eig(V);d=diag(D);d(d<=0)=10*realmin;
146        V= V3*diag(d)*V3';
147        L = tril(V,-1);V=diag(diag(V))+L+L'; % Check for symmetry again
148        V2=chol(V);
149    end
150end
151i=prerun+1;
152%% Burnin run
153while i <= burnin
154    if OPT.dsp % Display completion status
155        if mod(i,floor(burnin/20))==0
156            pcom=pcom+5;
157            fprintf('Complete = %d%% burn-in, Time since last update = %f s\n',pcom,cputime-mark); 
158            if i>2
159            remaining = (20 - i/floor(burnin/20))*(cputime-mark);
160            hrs = floor(remaining/3600); remaining = rem(remaining,3600);
161            mins = floor(remaining/60); 
162            secs = floor(rem(remaining,60));
163            fprintf('Predicted burn-in completion in %d hrs:%d mins:%d secs\n',hrs,mins,secs)
164            fprintf('Acceptance rate %f %%\n',100*avg_acc);
165            end;
166        mark=cputime;
167        end;
168    end; 
169    X_new = X + sqrt(s)*RD(i,:)*V2;     % propose a new sample
170    P_new = logpdf(X_new,OPT.data);              % new density
171    if P_new>best_P                     % new density is highest 
172        best_X=X_new;                   % record best seen parameters
173        best_P = P_new;
174        X = X_new;                      % accept X_new
175        P0 = P_new;
176        acc_rate=1;                     % record acceptance rate
177    else 
178        rho = P_new - P0;
179        acc_rate=exp(min(rho,0));       % record acceptance rate
180        if U(i)<= rho
181        X = X_new;                      % accept X_new 
182        P0 = P_new;
183        end
184    end
185    % Check stopping rule at every chk_stop counts
186    if ~mod(i-prerun,chk_stop)
187        if (log(mean(abs(delta(i-chk_stop+1:i-chk_stop/2,1)))) <= log(mean(abs(delta(i-chk_stop/2:i-1,1))))) &&...
188           (log(mean(abs(delta(i-chk_stop+1:i-chk_stop/2,2)))) <= log(mean(abs(delta(i-chk_stop/2:i-1,2)))))
189            burnin = i;
190            delta = delta(1:burnin,:);   % Adaptation cost
191        end
192    end
193    % Update scale
194    gamma = i^eps;
195    delta(i,1) = gamma*(acc_rate-optimal);
196    s = exp(log(s)+delta(i,1));
197    avg_acc = avg_acc*(i-1)/i + acc_rate/i;
198    % Update Sigma
199    if OPT.cov == 1
200        delta_X=(X-X_bar); X_bar = X_bar+gamma*delta_X;
201        Sdelta=gamma*(delta_X'*delta_X)-((i-1)^eps)*V;
202        delta(i,2) = norm(Sdelta,1);
203        V = V+Sdelta; L = tril(V,-1);V=diag(diag(V))+L+L'; % This help keeping V symmetrical
204        [V2,errV]=chol(V);
205        if errV
206            warning('Bad covariance matrix at burnin, perform reconditioning..')
207            [V3,D] = eig(V);d=diag(D);d(d<=0)=10*realmin;
208            V= V3*diag(d)*V3';
209            L = tril(V,-1);V=diag(diag(V))+L+L';
210            V2=chol(V);
211            i=i-1;
212        end
213    end
214    i=i+1;
215end
216G.cputime.burnin = cputime-mark_0; % Record time taken to adapt
217if OPT.dsp % If requested, give feedback on completion status
218    if OPT.cov == 1
219        figure;plot(1:burnin,delta(:,2));
220        title('Adaptation process of the covariance matrix $$\mathbf{\Sigma}$$');set(gca,'YScale','log');
221        xlabel('Number of iterations');ylabel('$$\Delta_m = \Vert\Sigma_m - \Sigma_{m-1}\Vert_1$$');
222        axis tight;drawnow
223    end
224    
225    figure;plot(1:burnin,abs(delta(:,1)));
226    title('Adaptation process of the scaling factor $$s$$');set(gca,'YScale','log');
227    xlabel('Number of iterations');ylabel('$$\Delta_m = \Vert s_m - s_{m-1}\Vert$$');
228    axis tight;drawnow
229    
230    fprintf('Burn-in completed after %d samples\n',burnin);
231    fprintf('Acceptance rate %f %%\n',100*avg_acc);
232end
233%% Final run
234div = ceil(M/1e5);              % Generate 'div' consecutive set of samples
235mrun = ceil(M/div);             % with 'mrun' samples in each set
236his = zeros(1.2*OPT.hgrd+1,Dims);   % Histogram grid points
237X_bar = zeros(size(X));         % Sample mean vector
238pcom = 0;                       % Completion percentage 
239mark = cputime;mark_0 = mark;   % Used to keep track of elapsed time
240for m=1:div                     % Devide run into 'div' number of pieces to save RAM
241    U = log(rand(mrun,1));      % Random acceptance numbers 
242    RD = randn(mrun,Dims)*V2;   % Random distubance proposal
243    TH = zeros(mrun,Dims);      % Final MCMC output
244    avg_acc = 0;                % Average acceptance rate
245    for i=1:mrun
246        X_new = X + sqrt(s)*RD(i,:);    % propose a new sample
247        P_new = logpdf(X_new,OPT.data);          % new density
248        if P_new>best_P                 % new density is highest 
249            best_X =X_new;              % record best seen parameters
250            best_P = P_new;
251            X      = X_new;             % accept X_new 
252            P0 = P_new;
253            acc_rate=1;                 % record acceptance rate
254        else 
255            rho = P_new - P0;
256            acc_rate=exp(min(rho,0));   % record acceptance rate
257            if U(i)<= rho
258            X = X_new;                  % accept X_new 
259            P0 = P_new;
260            end
261        end
262        TH(i,:) = X;
263        %Update scale
264        idx = (i+(m-1)*mrun+burnin);    % Global index
265        gamma=idx^eps;
266        s =exp(log(s)+gamma*(acc_rate-optimal));
267        avg_acc = avg_acc*(i-1)/i + acc_rate/i;
268        %Update Sigma
269        delta_X=X-X_bar;    X_bar = X_bar+gamma*delta_X;
270        Sdelta=gamma*(delta_X'*delta_X)-((idx-1)^eps)*V;
271        V = V+Sdelta;   L = tril(V,-1);V=diag(diag(V))+L+L';
272        [V2,errV]=chol(V);
273        if errV
274            warning('Bad covariance matrix at generation state, perform reconditioning..')
275            [V3,D] = eig(V);d=diag(D);d(d<=0)=10*realmin;
276            V= V3*diag(d)*V3';
277            L = tril(V,-1);V=diag(diag(V))+L+L';
278            V2=chol(V);
279        end
280    end;
281    %% Define grid points
282    h_max = max(TH);    h_min = min(TH);    
283    if m==1 % First grid
284        range = 1.2;
285        h_range = h_max-h_min;
286        hist_grid = my_linspace(h_min-0.1*h_range,h_max+0.1*h_range,range*OPT.hgrd+1);
287        mark_low = min(hist_grid,[],3); mark_high = max(hist_grid,[],3); 
288    else    % Subsequence grid
289        while any(h_max>mark_high) || any(h_min<mark_low)
290            % Expand grid points
291            range = range + 0.2;
292            hist_grid = my_linspace(mark_low-0.1*h_range,mark_high+0.1*h_range,range*OPT.hgrd+1);
293            mark_low = min(hist_grid,[],3); mark_high = max(hist_grid,[],3); 
294            % Added zeros into old histogram
295            his = [zeros(0.1*OPT.hgrd,Dims); his; zeros(0.1*OPT.hgrd,Dims)];
296        end
297    end
298    %% Compute raw sample histogram
299    for k=1:Dims
300        [memo,~] = hist(TH(:,k),hist_grid(1,k,:)); 
301        his(:,k)= his(:,k)*(m-1)/m+memo'/m;
302    end
303    %% Compute sample mean
304    X_bar = X_bar*(m-1)/m+mean(TH)/m;
305    %% Save MCMC samples to hard disk and free RAM
306    save([OPT.filename '_' num2str(m)],'TH','-v7.3')
307    if OPT.dsp  % Display completion status
308        pcom=pcom+100/div;
309        fprintf('Complete = %d%%, Time since last update = %f s\n',pcom,cputime-mark); 
310        remaining = (div - m)*(cputime-mark);
311        hrs = floor(remaining/3600); remaining = rem(remaining,3600);
312        mins = floor(remaining/60); 
313        secs = floor(rem(remaining,60));
314        fprintf('Predicted completion in %d hrs:%d mins:%d secs\n',hrs,mins,secs)
315        fprintf('Acceptance rate %f %%\n',100*avg_acc);
316        mark=cputime;
317    end
318end
319G.div = div;
320G.cputime.run = cputime-mark_0;         % Record time taken to run MCMC
321G.best_TH = best_X;                     % Record highest density state
322G.mean = X_bar;                         % Sample mean state
323G.filename = OPT.filename;              % Passing back the filename
324% Normalise sample histograms in each dimension
325for k=1:Dims
326    h.p = his(:,k);h.x = hist_grid(1,k,:);h.x = h.x(:);
327    h.p = h.p/(sum(h.p)*(h.x(2)-h.x(1)));
328    G.hist.p(:,k) = h.p; G.hist.x(:,k) = h.x;
329end
330if OPT.dsp  % Display final results
331    fprintf('Successfully generated %d samples in %d seconds\n',div*mrun, G.cputime.run);
332end
